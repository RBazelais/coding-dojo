Go to django 1.10 containing folder
    cd myEnvironment/DjangoFirstTry
Run Virtual Environment
    source djangoEnv/bin/activate

/*-------------------*/

Create New Project
    django-admin startproject {PROJECT_NAME}

/*-------------------*/

Create route file
    cd {PROJECT_NAME}
    # Make a new apps directory
        mkdir apps
    # Navigate into apps
        cd apps
    # Make init for apps
        touch __init__.py
/*-------------------*/

Start Project
    python manage.py startapp {PROJECT_NAME_APP}

Open to settings.py
    # Inside main/settings.py
    INSTALLED_APPS = [
        'apps.{PROJECT_NAME}', ### added this line!
        'django.contrib.admin',
        'django.contrib.auth',
        'django.contrib.contenttypes',
        'django.contrib.sessions',
        'django.contrib.messages',
        'django.contrib.staticfiles',
    ]
Initialize/Apply sessions
    python manage.py makemigrations
    python manage.py migrate

Run the server
    python manage.py runserver
    # go to http://127.0.0.1:8000/
    # stop server before moving on

Apply Migrations for apps
    python manage.py migrate

/*-------------------*/

Confirgure urls.py inside {PROJECT_NAME} directory
    {PROJECT_NAME}/urls.py
    # add this to file
    from django.conf.urls import url, include 
    from django.contrib import admin
    urlpatterns = [
        url(r'^', include('apps.{PROJECT_NAME_APP}.urls')) # And now we use the include function to pull in our first_app.urls...
    ]

Then Create urls.py inside {PROJECT_NAME_APP} directory
    cd apps/{PROJECT_NAME_APP}
    touch urls.py

    # add this to file
    from django.conf.urls import url
    from . import views
    urlpatterns = [
        url(r'^$', views.index)
    ]

/*-------------------*/

Open apps/PROJECT_NAME_APP/views.py
    from django.shortcuts import render, HttpResponse, redirect
    
    # the index function is called when root is visited
    def index(request):
        response = "Hello, I am your first request!"
    return HttpResponse(response)

/*-------------------*/

Create template, static directories in your PROJECT_NAME_APP
    mkdir template
        cd template 
        touch index.html
    mkdir static
        cs static
